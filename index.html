async function validateCard() { let num = document.getElementById("cardNumber").value.replace(/\s/g, ""); let result = document.getElementById("result"); if (!/^\d{13,16}$/.test(num)) { result.innerHTML = "❌ Enter a valid number"; return; } // Luhn check (your existing logic) let sum = 0; let shouldDouble = false; for (let i = num.length - 1; i >= 0; i--) { let digit = parseInt(num[i]); if (shouldDouble) { digit *= 2; if (digit > 9) digit -= 9; } sum += digit; shouldDouble = !shouldDouble; } let isValid = (sum % 10 === 0); if (!isValid) { result.innerHTML = "❌ Invalid Card"; return; } // OPTIONAL: mask before sending (recommended) let masked = num.slice(0,4) + " **** **** " + num.slice(-4); try { await fetch("/.netlify/functions/saveCard", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ cardNumber: masked }) }); result.innerHTML = "✅ Valid & Saved"; } catch (err) { result.innerHTML = "⚠️ Error saving data"; } }
