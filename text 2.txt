<script>
function validateCard() {
  let num = document.getElementById("cardNumber").value.replace(/\s+/g, '');
  let result = document.getElementById("result");

  if (!num) {
    result.innerHTML = "Please enter a card number";
    return;
  }

  // Luhn Algorithm
  let sum = 0;
  let shouldDouble = false;

  for (let i = num.length - 1; i >= 0; i--) {
    let digit = parseInt(num[i]);

    if (shouldDouble) {
      digit *= 2;
      if (digit > 9) digit -= 9;
    }

    sum += digit;
    shouldDouble = !shouldDouble;
  }

  let isValid = (sum % 10 === 0);

  // Card type detection
  let type = "Unknown";
  if (/^4/.test(num)) type = "Visa";
  else if (/^5[1-5]/.test(num)) type = "MasterCard";
  else if (/^3[47]/.test(num)) type = "American Express";

  result.innerHTML = isValid
    ? `✅ Valid Card <br> Type: ${type}`
    : `❌ Invalid Card`;
}
</script>
